1. Get the application URL:
{{- if .Values.ingress.enabled }}
  Visit http://{{ .Values.ingress.host }}
{{- else }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Values.namespace }} -l "app.kubernetes.io/name={{ include "localstore.fullname" . }}-frontend" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Values.namespace }} port-forward $POD_NAME 8080:80
  Visit http://localhost:8080
{{- end }}

2. Seed the database (first install only):
  kubectl exec -it deploy/{{ include "localstore.fullname" . }}-backend -n {{ .Values.namespace }} -- node prisma/seed.js
